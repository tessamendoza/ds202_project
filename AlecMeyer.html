What are the different ways people spend time?

time1 <- time %>%
  group_by(WBLADDER, ACT_FOOD, ACT_PCARE, ACT_RELIG, ACT_SOCIAL, ACT_SPORTS, ACT_TRAVEL, ACT_WORK) %>%
  summarise(num = n()) %>%
  filter(WBLADDER != 999) 
## `summarise()` regrouping output by 'WBLADDER', 'ACT_FOOD', 'ACT_PCARE', 'ACT_RELIG', 'ACT_SOCIAL', 'ACT_SPORTS', 'ACT_TRAVEL' (override with `.groups` argument)
time1$WBLADDER <- factor(time1$WBLADDER, labels = c("Worst", "Horrible", "Bad", "Sucks", "Alright", "Fine", "Good", "Great", "Amazing", "The greatest", "Best"))
